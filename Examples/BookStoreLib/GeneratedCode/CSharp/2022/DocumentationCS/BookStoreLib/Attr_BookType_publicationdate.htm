LiquidTechnologies.Runtime.XmlDateTime Publicationdate
  Property Description  
    Property Name Publicationdate  
    Property Type LiquidTechnologies.Runtime.XmlDateTime  
    Parent Class BookStoreLib.BookType  
    XML Type Attribute  
    Accessors Read/Write  
    Description An optional LiquidTechnologies.Runtime.XmlDateTime  
    Remarks It is optional, initially it is not populated.


  
    Example 
BookStoreLib.BookType oParent = ...;

// Because the property is Optional, it will NOT Initially be populated
// Attempting to read from it before it is set WILL RAISE AN EXCEPTION
// You can tell whether it is populated by checking 
// the IsValidPublicationdate flag.
System.Diagnostics.Debug.Assert(oParent.IsValidPublicationdate == false);
      
// So it can be set as follows 
LiquidTechnologies.Runtime.XmlDateTime oNewItem = new LiquidTechnologies.Runtime.XmlDateTime();
oNewItem.SetDate(2005, 11, 29);
oParent.Publicationdate = oNewItem;
      
// As the property is populated IsValidPublicationdate will now return true
System.Diagnostics.Debug.Assert(oParent.IsValidPublicationdate == true);
      
// So it can be read  as follows 
System.Diagnostics.Debug.WriteLine(oParent.Publicationdate.ToString());
