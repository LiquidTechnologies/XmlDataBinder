string Genre
  Property Description  
    Property Name Genre  
    Property Type string  
    Parent Class BookStoreLib.BookType  
    XML Type Element  
    Accessors Read/Write  
    Description An optional string  
    Remarks It is optional, initially it is not populated.


  
    Example 
BookStoreLib.BookType oParent = ...;

// Because the property is Optional, it will NOT Initially be populated
// Attempting to read from it before it is set WILL RAISE AN EXCEPTION
// You can tell whether it is populated by checking 
// the IsValidGenre flag.
System.Diagnostics.Debug.Assert(oParent.IsValidGenre == false);
      
// So it can be set as follows
oParent.Genre = "Sample Value";
      
// As the property is populated IsValidGenre will now return true
System.Diagnostics.Debug.Assert(oParent.IsValidGenre == true);
      
// So it can be read  as follows 
System.Diagnostics.Debug.WriteLine(oParent.Genre.ToString());
